def processar_dnit(df):

    #Removendo campos nulos
    df_dnit = df[(df['GeocodOrigem'].notna()) & (df['GeocodDestino'].notna())]
    df_dnit = df_dnit.astype({"GeocodOrigem": int, "GeocodDestino": int})

    # #Filtrando observaes por tipo de automvel
    df_dnit = df_dnit[(df_dnit['Classe']=="P2") 
             | (df_dnit['Classe']=="O1") 
             | (df_dnit['Classe']=="O2") 
             | (df_dnit['Classe']=="O3")] 
    
    df_dnit = pd.DataFrame({'freq_semanal' : df_dnit.groupby( [ "GeocodOrigem", "GeocodDestino" ] ).size()}).reset_index()
    
    #Removendo ligaes que possuem mesma origem e destino
    df_dnit = df_dnit[df_dnit["GeocodOrigem"]!=df_dnit["GeocodDestino"]]
    return df_dnit

def adicionar_estado(df_estado, df_dnit):
    df_dnit["estado_origem"] = df_dnit["GeocodOrigem"].astype(str).str[:2]
    df_dnit["estado_origem"] = df_dnit["estado_origem"].astype(int)
    df_dnit["estado_destino"] = df_dnit["GeocodDestino"].astype(str).str[:2]
    df_dnit["estado_destino"] = df_dnit["estado_destino"].astype(int)

    df_dnit = df_dnit.merge(df_estado[["cod_uf", "uf"]], left_on="estado_origem", right_on= "cod_uf")
    df_dnit.rename(columns={"uf": "uf_origem"}, inplace=True)
    df_dnit = df_dnit.merge(df_estado[["cod_uf", "uf"]], left_on="estado_destino", right_on= "cod_uf")
    df_dnit.rename(columns={"uf": "uf_destino"}, inplace=True)

    return df_dnit

# Adiciona o codigo do arranjo no municpio IBGE e DNIT
def adiciona_cod_arranjo(df_municipios, df):
    df = df.merge(df_municipios[['cod_mun','cod_arranjo', 'compoe_arranjo']], left_on=['origem'], right_on = ['cod_mun'])
    df = df.drop(columns=['cod_mun'])
    df.rename(columns={"cod_arranjo": "cod_arranjo_origem", "compoe_arranjo": "compoe_arranjo_ori"}, inplace=True)
    df = df.merge(df_municipios[['cod_mun','cod_arranjo', 'compoe_arranjo']], left_on=['destino'], right_on = ['cod_mun'])
    df.rename(columns={"cod_arranjo": "cod_arranjo_destino", "compoe_arranjo": "compoe_arranjo_dest"}, inplace=True)
    df = df.drop(columns=['cod_mun'])
    return df
